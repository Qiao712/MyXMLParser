93 inline uint32_t StringUtility::strToIntHex(const char* beg, const char* end)
94 {
95 if (beg >= end) return static_cast<uint16_t>(-1);
96 uint32_t result = 0;
97 uint32_t x;
98 for (const char* p = beg; p < end; p++) {
99 if (*p >= 'a' && *p <= 'f') x = *p - 'a' + 10;
100 else if (*p >= 'A' && *p <= 'F') x = *p - 'A' + 10;
101 else if (*p >= '0' && *p <= '9') x = *p - '0';
102 else return static_cast<uint16_t>(-1);
103 result = result * 16 + x;
104 }
105
106 return result;
107 }

123 inline bool StringUtility::unicodeToUtf8(uint32_t unicode, std::string& utf8)
124 {
125 constexpr uint32_t MARKS[6] = {
126 ~static_cast<uint32_t>(0x007F),
127 ~static_cast<uint32_t>(0x07FF),
128 ~static_cast<uint32_t>(0xFFFF),
129 ~static_cast<uint32_t>(0x1FFFFF),
130 ~static_cast<uint32_t>(0x3FFFFFF),
131 ~static_cast<uint32_t>(0x7FFFFFFF)
132 };
133 constexpr uint32_t FIRST_BYTE[] = {
134 0x00,
135 0xC0,
136 0xE0,
137 0xF0,
138 0xF8,
139 0xFC
140 };
141 constexpr uint32_t LOWER_BYTE = 0x80; //1000 0000
142 constexpr uint32_t LOWER_6BITS = 0x3F; //0011 1111
143 constexpr uint32_t MAX_UNICODE = 0x10FFFF;
144 static char buf[6];
145
146 if (unicode > MAX_UNICODE) return false;
147
148 int n = 0;
149 while (unicode & MARKS[n]) n++;
150
151 for (int i = n; i > 0; i--) {
152 //10xxxxxx
153 buf[i] = static_cast<char>(LOWER_BYTE | (unicode & LOWER_6BITS));
154 unicode >>= 6;
155 }
156 buf[0] = static_cast<char>(FIRST_BYTE[n] | unicode);
157
158 utf8.assign(buf, buf + n + 1);
159
160 return true;
161 }
